    function funClick(button) {
      let messages = [
        "You found a hidden coin!",
        "Surprise! A funny fact: Cats sleep 70% of their lives.",
        "Boom! You earned 10 imaginary points!",
        "Did you know? The shortest war in history lasted 38 minutes!"
      ];
      let colors = ["#ff6b6b", "#6bc1ff", "#ffd36b", "#a56bff"];
      let msgElement = document.getElementById('msg' + button);
      let randomIndex = Math.floor(Math.random() * messages.length);
      msgElement.innerText = messages[randomIndex];
      msgElement.style.color = colors[randomIndex];
    }
